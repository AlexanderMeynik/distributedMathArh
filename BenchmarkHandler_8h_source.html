26template<typename...ARRAYS>
27constexpr auto CartesianProduct(ARRAYS...arrays) {
28 using Type = std::tuple<typename ARRAYS::value_type...>;
29
30 constexpr std::size_t kN = (1 * ... * arrays.size());
31
32 std::array<std::size_t, sizeof...(arrays)> dims{arrays.size()...};
33 for (std::size_t i = 1; i < dims.size(); ++i) { dims[i] *= dims[i - 1]; }
34
35 return [&]() {
36 std::array<Type, kN> result;
37
38 for (std::size_t i = 0; i < result.size(); ++i) {
39 [&]<std::size_t... Is>(std::index_sequence<Is...>) {
40 auto idx = std::make_tuple(((i * std::get<Is>(dims)) / kN) % arrays.size() ...);
41 result[i] = std::make_tuple(arrays[std::get<Is>(idx)]...);
42
43 }(std::make_index_sequence<sizeof...(arrays)>{});
44 }
45
46 return result;
47 }();
48}
