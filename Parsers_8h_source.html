118Struct JsonToContinuous(const Json::Value &val,
119 std::optional<size_t> sz,
120 bool sizeless_format) {
121
122 size_t size;
123 if(sz.has_value())
124 {
125 size=sz.value();
126 }
127 else {
128 if (sizeless_format) {
129 size = val.size();
130 } else {
131 size = val["size"].asUInt();
132 }
133 }
134
135 Struct res(size);
136 if(!sizeless_format)
137 {
138 for (int i = 0; i < size; ++i) {
139 res[i] = val["data"][i].as<std::remove_all_extents_t<typename Struct::value_type>>();
140 }
141 }
142 else
143 {
144 for (int i = 0; i < size; ++i) {
145 res[i] = val[i].as<std::remove_all_extents_t<typename Struct::value_type>>();
146 }
147 }
148
149 return res;
150}

177Struct ParseSolution(std::istream &in,
178 IoFormat format,
179 std::optional<size_t> size_opt,
180 const EFormat &ef) {
181
182 Struct res;
183 switch (format) {
184 case IoFormat::SERIALIZABLE:return ParseOneDim<Struct>(in, size_opt, ef);
185 case IoFormat::HUMAN_READABLE: {
186 size_t n;
187 in >> n;
188 std::string dummy;
189 std::getline(in, dummy);
190 std::getline(in, dummy);
191 std::getline(in, dummy);
192
193 if (!in) {
194 throw ioError(to_string(in.rdstate()));
195 }
196
197 Struct sol(n * 4);
198
199 for (int i = 0; i < n; i++) {
200 FloatType ax, bx, cx;
201 FloatType ay, by, cy;
202 in >> ax >> bx >> cx >> ay >> by >> cy;
203
204 sol[2 * i] = ax;
205 sol[2 * i + 2 * n] = bx;
206 sol[2 * i + 1] = ay;
207 sol[2 * i + 1 + 2 * n] = by;
208
209 }
210 return sol;
211
212 }
213 }
214 return res;
