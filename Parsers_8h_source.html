41

44 in.get();

48 for (int i = 0; i < N; ++i) {
49

51 avec[j][i + N] = lim[2];
52 lim[0] += step[0];

66 avec[j][i * N2 + k] = lim[0];
67 avec[j][i * N2 + k + N] = lim[2];
68 lim[2] += step[1];
69 }
70 lim[2] = start[1];
71 lim[0] += step[0];
72 }
73 } else {
74 for (int i = 0; i < N; ++i) {
75 in >> avec[j][i];
