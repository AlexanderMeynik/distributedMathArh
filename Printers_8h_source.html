193Json::Value ContinuousToJson(const Struct &col,
194 bool print_size,
195 bool sizeless_format) {
196 Json::Value res;
197 if (!sizeless_format) {
198 if (print_size) {
199 res["size"] = col.size();
200 }
201 for (size_t i = 0; i < col.size(); i++) {
202 res["data"][(Json::ArrayIndex) i] = col[i];
203 }
204 } else {
205 for (size_t i = 0; i < col.size(); i++) {
206 res[(Json::ArrayIndex) i] = col[i];
207 }
208 }
209
210 return res;
