176 Json::Value res;
177 if(printSize) {
178 res["size"] = col.size();

198
199 for (int i = 0; i < N_; i++) {
200 auto cx = sqrt(solution[2 * i] * solution[2 * i] +
201 solution[2 * i + 2 * N_] * solution[2 * i + 2 * N_]);
202 auto cy = sqrt(solution[2 * i + 1] * solution[2 * i + 1] +
203 solution[2 * i + 1 + 2 * N_] * solution[2 * i + 1 + 2 * N_]);
204
205 out << solution[2 * i] << "\t"
206 << solution[2 * i + 2 * N_] << "\t"
207 << cx << "\n";
208 out << solution[2 * i + 1] << "\t"
209 << solution[2 * i + 1 + 2 * N_] << "\t"
210 << cy << "\n";
