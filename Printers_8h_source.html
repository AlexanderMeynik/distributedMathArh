170 out << "Решение системы диполей\n Ai(x\\ny)\tBi(x\\ny)\tCi(x\\ny)\n";
171 int N_ = solution.size() / 4.0;
172
173 for (int i = 0; i < N_; i++) {
174 auto cx = sqrt(solution[2 * i] * solution[2 * i] +
175 solution[2 * i + 2 * N_] * solution[2 * i + 2 * N_]);
176 auto cy = sqrt(solution[2 * i + 1] * solution[2 * i + 1] +
177 solution[2 * i + 1 + 2 * N_] * solution[2 * i + 1 + 2 * N_]);
178
179 out << solution[2 * i] << "\t"
180 << solution[2 * i + 2 * N_] << "\t"
181 << cx << "\n";
182 out << solution[2 * i + 1] << "\t"
183 << solution[2 * i + 1 + 2 * N_] << "\t"
184 << cy << "\n";
185
186 }
187 return 0;
188 }
