189Json::Value ContinuousToJson(const Struct &col,
190 bool print_size,
191 bool sizeless_format) {
192 Json::Value res;
193 if (!sizeless_format) {
194 if (print_size) {
195 res["size"] = col.size();
196 }
197 for (size_t i = 0; i < col.size(); i++) {
198 res["data"][(Json::ArrayIndex) i] = col[i];
199 }
200 } else {
201 for (size_t i = 0; i < col.size(); i++) {
202 res[(Json::ArrayIndex) i] = col[i];
203 }
204 }
205
206 return res;
207}
