208 template<template<typename ...> typename container, typename T, bool end>
209 container<T> myLinspace(T lower_bound, T upper_bound, size_t n) {
210
211 if (n == 0 || lower_bound-upper_bound==0) {
212 throw std::invalid_argument("Zero linspace size");
213 }
214
215 container<T> result(n);
216
217 size_t div;
218 if constexpr (end) {
219 div = n - 1;
220 } else {
221 div = n;
222 }
223 T step = (upper_bound - lower_bound) / (T) div;
224
225 for (int i = 0; i < n; ++i) {
226 result[i] = i * step + lower_bound;
227 }
228 return result;
229 }
