225 container<T> myLinspace(T lower_bound, T upper_bound, size_t n) {
226
227 if (n == 0 || lower_bound - upper_bound == 0) {
228 throw std::invalid_argument("Zero linspace size");
229 }
230
231 container<T> result(n);
232
233 size_t div;
234 if constexpr (end) {
235 div = n - 1;
236 } else {
237 div = n;
238 }
239 T step = (upper_bound - lower_bound) / (T) div;
240
241 for (int i = 0; i < n; ++i) {
242 result[i] = i * step + lower_bound;
