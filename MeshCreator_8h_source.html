201 template<template<typename ...> typename container, typename T, bool end>
202 container<T> myLinspace(T lower_bound, T upper_bound, size_t n) {
203
204 if (n == 0 || lower_bound - upper_bound == 0) {
205 throw std::invalid_argument("Zero linspace size");
206 }
207
208 container<T> result(n);
209
210 size_t div;
211 if constexpr (end) {
212 div = n - 1;
213 } else {
214 div = n;
215 }
216 T step = (upper_bound - lower_bound) / (T) div;
217
218 for (int i = 0; i < n; ++i) {
219 result[i] = i * step + lower_bound;
220 }
221 return result;
222 }
