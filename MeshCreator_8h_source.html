213template<template<typename ...> typename container, typename T, bool end>
214container<T> MyLinspace(T lower_bound, T upper_bound, size_t n) {
215
216 if (n == 0 || lower_bound - upper_bound == 0) {
217 throw std::invalid_argument("Zero linspace size");
218 }
219
220 container<T> result(n);
221
222 size_t div;
223 if constexpr (end) {
224 div = n - 1;
225 } else {
226 div = n;
227 }
228 T step = (upper_bound - lower_bound) / (T) div;
229
230 for (int i = 0; i < n; ++i) {
231 result[i] = i * step + lower_bound;
232 }
233 return result;
234}

238MyMeshGrid(const container<co::FloatType> &a,
239 const container<co::FloatType> &b) {

242
243 for (size_t i = 0; i < b.size(); ++i) {
244 for (size_t j = 0; j < a.size(); ++j) {
245 ret[0][i * a.size() + j] = a[j];
246 ret[1][i * a.size() + j] = b[i];
