249 clockArray<OutType, inType, timeGetter, sourceTypeConverter, timeConverter>::tak(

251 const guardType guard{s_mutex};
252 auto id = (*sourceTypeConverter)(location);
253 if (toTak.empty() || toTak.top()[0] != id[0]) {
254 std::string msg = "No paired tik statement found in queue\t"
255 "Tak values" + id[3] + ":" + id[1] + "\t"
256 + id[0] + '\t' + id[2];
257 throw std::logic_error(msg);
258 }
259 id[1] = toTak.top()[1];
260 id[2] = toTak.top()[2];
261 toTak.pop();
262 auto res = timeConverter((*timeGetter)(), startIngTimers[id]);
263 if (!timers.contains(id)) {
264 timers[id] = {res, 1};
265 } else {
266 timers[id].time += res;
267 timers[id].count++;
268 }
