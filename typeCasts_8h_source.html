38 std::array<std::vector<T>, 2> reinterpretVector(Eigen::Vector<T, Eigen::Dynamic> &xi) {
39  auto N = xi.size() / 2;
40  if (!N) {
41  return std::array<std::vector<T>, 2>();
42  }
43  std::array<std::vector<T>, 2> res;
44  res[0] = std::vector<T>(N, 0);
45  res[1] = std::vector<T>(N, 0);
46 
47  for (int i = 0; i < N; ++i) {
48  res[0][i] = xi[i];
49  res[1][i] = xi[i + N];
