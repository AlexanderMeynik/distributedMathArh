37template<class T>
38std::array<std::vector<T>, 2> reinterpretVector(Eigen::Vector<T, Eigen::Dynamic> &xi) {
39 auto N = xi.size() / 2;
40 if (!N) {
41 return std::array<std::vector<T>, 2>();
42 }
43 std::array<std::vector<T>, 2> res;
44 res[0] = std::vector<T>(N, 0);
45 res[1] = std::vector<T>(N, 0);
46
47 for (int i = 0; i < N; ++i) {
48 res[0][i] = xi[i];
49 res[1][i] = xi[i + N];
50 }
51 return res;
52}

121 , Eigen::Dynamic,Eigen::Dynamic>>
122toEigenMatrix(Container&container, int columns)
123{
124 if(container.size() == 0) {
125 throw std::length_error("Zero input container size!");
126 }
127
128 if(container.size() % columns != 0) {
129 throw std::length_error("Invalid solutions size to columns ratio: container.size() % columns = "+std::to_string(container.size() % columns));
130 }
131 size_t rows= container.size() / columns;
132
133
134 Eigen::Map<Eigen::Matrix<std::remove_reference_t<typename Container::value_type>
135 , Eigen::Dynamic,Eigen::Dynamic>> map(&container[0], rows, columns);
136
137
138 return map;
