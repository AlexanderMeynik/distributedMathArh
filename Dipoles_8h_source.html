148template<typename Container>
149void Dipoles::SetMatrixes(const Container &xi) {
150 std::vector<std::pair<int, int>> sectors(4);
151 sectors[0] = {0, 0};
152 sectors[1] = {0, 2 * n_};
153 sectors[2] = {2 * n_, 0};
154 sectors[3] = {2 * n_, 2 * n_};
155 for (int i = 0; i < n_; ++i) {//MAC
156 for (int m = 0; m < n_; ++m) {
157 if (i == m) {
158 Eigen::Matrix<FloatType, 2, 2> id = Eigen::Matrix<FloatType, 2, 2>::Identity() *
159 (params::omega0 * params::omega0 -
160 params::omega * params::omega);
161 m_1_.template block<2, 2>(2 * i, 2 * m) = id;
162 } else {
163
164 Eigen::Vector<FloatType, 2> rim = TwoDVecDifference(m, i, xi);
165 auto r_mode = GetDistance(i, m, xi);
166 Eigen::Matrix<FloatType, 2, 2> k1;
167 Eigen::Matrix<FloatType, 2, 2> k2;
168 GetMatrices(rim, r_mode, k1, k2);
169 auto arg = params::omega * r_mode / params::c;
170 Eigen::Matrix<FloatType, 2, 2> tmpmatr = -an_ * (k1 * cos(arg) - k2 * sin(arg));
171 m_1_.template block<2, 2>(2 * i, 2 * m) = tmpmatr;
172 }
173 }
174 }
175
176 for (int i = 0; i < n_; ++i) {//MBC
177 for (int m = 0; m < n_; ++m) {
178 if (i == m) {
179 Eigen::Matrix<FloatType, 2, 2> id =
180 Eigen::Matrix<FloatType, 2, 2>::Identity() * (params::yo * params::omega);
181 m_2_.template block<2, 2>(2 * i, 2 * m) = -id;
182 } else {
183
184 Eigen::Vector<FloatType, 2> rim = TwoDVecDifference(m, i, xi);
185 auto r_mode = GetDistance(i, m, xi);
186 Eigen::Matrix<FloatType, 2, 2> k1;
187 Eigen::Matrix<FloatType, 2, 2> k2;
188 GetMatrices(rim, r_mode, k1, k2);
189 auto arg = params::omega * r_mode / params::c;
190
191 Eigen::Matrix<FloatType, 2, 2> tmpmatr = -an_ * (k2 * cos(arg) + k1 * sin(arg));
192
193 m_2_.template block<2, 2>(2 * i, 2 * m) = tmpmatr;

212void Dipoles::GetFullFunction(const Container &xi, const Container2 &sol) {
213 this->ifunction_ = [&xi, &sol](FloatType theta, FloatType phi, FloatType t) {
214 int n = xi.size() / 2;
215 FloatType res = 0;
216 FloatType s[2] = {cos(phi), sin(phi)};
217 FloatType ress[3] = {0, 0, 0};
218 for (int i = 0; i < n; ++i) {
219 FloatType ri[2] = {GetElement(xi, 0, i, n),
220 GetElement(xi, 1, i, n)};
221 FloatType ys = (ri[1] * cos(phi) - ri[0] * sin(phi)) * sin(theta);
222 FloatType t0 = t - ys / params::c;
223 FloatType aI[2] = {GetValue(sol, 2 * i), GetValue(sol, 2 * i + 1)};
224 FloatType bi[2] = {GetValue(sol, 2 * i + 2 * n), GetValue(sol, 2 * i + 1 + 2 * n)};
225
226 FloatType di[2] = {aI[0] * cos(params::omega * t0) + bi[0] * sin(params::omega * t0),
227 aI[1] * cos(params::omega * t0) + bi[1] * sin(params::omega * t0)};
228
229 FloatType vi[2] = {params::omega * (bi[0] * cos(params::omega * t0)
230 - aI[0] * sin(params::omega * t0)) / params::c,
231 params::omega * (bi[1] * cos(params::omega * t0)
232 - aI[1] * sin(params::omega)) / params::c};
233 FloatType ai[2] = {-pow(params::omega, 2) * di[0], -pow(params::omega, 2) * di[1]};
234
235 FloatType vsi = vi[0] * s[0] + vi[1] * s[1];
236 FloatType asi = ai[0] * s[0] + ai[1] * s[1];
237
238 for (int coord = 0; coord < 2; ++coord) {
239 FloatType ttt = ai[coord] * (vsi * sin(theta) - 1) +
240 s[coord] * asi * pow(sin(theta), 2) -
241 vi[coord] * asi * sin(theta);
242 ress[coord] += ttt;
243 }
244
245 FloatType t3 = asi * sin(theta) * cos(theta);
246 ress[2] += t3;
247 }
248 for (FloatType elem : ress) {
249 res += elem * elem;
250 }
251 return res;
252 };

254 this->i_2_function_ = [&xi, &sol](FloatType phi, FloatType theta) {
255 int n = xi.size() / 2;
256 FloatType omega0 = params::omega;
257 FloatType t_0 = M_PI * 2 / omega0;
258 FloatType res;
259 Eigen::Vector<FloatType, 2> resxy = {0.0, 0.0};
260 FloatType resz = 0.0;
261 FloatType o3dco3dc = pow(omega0, 3) / params::c;//todo why is this here?
262 FloatType o2 = pow(omega0, 2);
263 FloatType sinth2 = pow(sin(theta), 2);
264 Eigen::Vector<FloatType, 2> s = {cos(phi),
265 sin(phi)};
266 for (int i = 0; i < n; ++i) {
267 FloatType ri[2] = {GetElement(xi, 0, i, n), GetElement(xi, 1, i, n)};
268 FloatType ys = (ri[1] * cos(phi) - ri[0] * sin(phi)) * sin(theta);
269 Eigen::Vector<FloatType, 2> Ai = {GetValue(sol, 2 * i), GetValue(sol, 2 * i + 1)};
270 Eigen::Vector<FloatType, 2> bi = {GetValue(sol, 2 * i + 2 * n), GetValue(sol, 2 * i + 1 + 2 * n)};
271
272 FloatType argument = omega0 * ys / params::c;
273 FloatType ais = Ai.dot(s);
274 FloatType bis = bi.dot(s);
275 Eigen::Vector<FloatType, 2> a_bis = Ai * bis;
276 Eigen::Vector<FloatType, 2> b_ais = bi * ais;
277
278 Eigen::Vector<FloatType, 2> pc_1_i = ((s * ais * sinth2 - Ai) * cos(argument) -
279 (s * bis * sinth2 - bi) * sin(argument));
280 Eigen::Vector<FloatType, 2> ps_1_i = ((s * ais * sinth2 - Ai) * sin(argument) +
281 (s * bis * sinth2 - bi) * cos(argument));
282 Eigen::Vector<FloatType, 2> pcomi = -(omega0 / params::c) * (sin(theta)) * (a_bis - b_ais);
283
284 FloatType pci = (ais * cos(argument) - bis * sin(argument));
285 FloatType psi = (ais * sin(argument) + bis * cos(argument));
286
287 Eigen::Vector<FloatType, 2> rj;
288 FloatType ysj;
289 Eigen::Vector<FloatType, 2> aj;
290 Eigen::Vector<FloatType, 2> bj;
291 FloatType argumentj;
292
293 FloatType ajs;
294 FloatType bjs;

296 FloatType pcj;
297 FloatType psj;

301 Eigen::Vector<FloatType, 2> pcomj;
302
303 Eigen::Vector<FloatType, 2> a_bjs;
304 Eigen::Vector<FloatType, 2> b_ajs;
305 for (int j = 0; j < i; ++j) {
306 rj = {GetElement(xi, 0, j, n), GetElement(xi, 1, j, n)};
307 ysj = (rj[1] * cos(phi) - rj[0] * sin(phi)) * sin(theta);
308
309 aj = {GetValue(sol, 2 * j), GetValue(sol, 2 * j + 1)};
310 bj = {GetValue(sol, 2 * j + 2 * n), GetValue(sol, j + 1 + 2 * n)};
311
312 argumentj = omega0 * ysj / params::c;
313 ajs = aj.dot(s);
314 bjs = bj.dot(s);
315
316 a_bjs = aj * bjs;
317 b_ajs = bi * ajs;
318
319 pcj = (ajs * cos(argumentj) - bjs * sin(argumentj));
320 psj = (ajs * sin(argumentj) + bjs * cos(argumentj));
321
322 pc_1_j = ((s * ajs * sinth2 - aj) * cos(argumentj) -
323 (s * bjs * sinth2 - bj) * sin(argumentj));
324 ps_1_j = ((s * ajs * sinth2 - aj) * sin(argumentj) +
325 (s * bjs * sinth2 - bj) * cos(argumentj));
326 pcomj = -(omega0 / params::c) * (sin(theta)) * (a_bjs - b_ajs);
327
328 Eigen::Vector<FloatType, 2> rij_xy =
329 pc_1_i.cwiseProduct(pc_1_j) + ps_1_i.cwiseProduct(ps_1_j)
330 + 2 * pcomi.cwiseProduct(pcomj);
331
332 FloatType rij_z = (pci * pcj + psi * psj);
333 resz += rij_z;
334 resxy += rij_xy;
335 }
336 Eigen::Vector<FloatType, 2> ri_xy =
337 (pc_1_i.cwiseProduct(pc_1_i) + ps_1_i.cwiseProduct(ps_1_i)
338 + 2 * pcomi.cwiseProduct(pcomi)) / 2.0;
339 resxy += ri_xy;
340
341 FloatType ri_xz = (pci * pci + psi * psi) / 2;
342 resz += ri_xz;
343 }
344 resxy = resxy * t_0 * o2 * o2;
345 resz = resz * t_0 * (o2 * o2 * pow(sin(theta) * cos(theta), 2));
346 res = resxy.sum() + resz;
347 return res;

355FloatType Dipoles::GetDistance(size_t i1, size_t i2, Container &xi) {
356 FloatType d1;
357 FloatType d2;
358 if constexpr (my_concepts::HasBracketsNested<Container>) {
359 d1 = xi[0][i1] - xi[0][i2];
360 d2 = xi[1][i1] - xi[1][i2];
361 } else {
362 d1 = xi[i1] - xi[i2];
363 d2 = xi[i1 + n_] - xi[i2 + n_];
364 }
365 return std::sqrt(d1 * d1 + d2 * d2);
366}
