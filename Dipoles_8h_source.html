217  this->ifunction_ = [&xi, &sol](FloatType theta, FloatType phi, FloatType t) {
218  int n = xi.size() / 2;
219  FloatType res = 0;
220  FloatType s[2] = {cos(phi), sin(phi)};
221  FloatType ress[3] = {0, 0, 0};
222  for (int i = 0; i < n; ++i) {
223  FloatType ri[2] = {GetElement(xi, 0, i, n),
224  GetElement(xi, 1, i, n)};
225  FloatType ys = (ri[1] * cos(phi) - ri[0] * sin(phi)) * sin(theta);
226  FloatType t0 = t - ys / params::c;
227  FloatType aI[2] = {GetValue(sol, 2 * i), GetValue(sol, 2 * i + 1)};
228  FloatType bi[2] = {GetValue(sol, 2 * i + 2 * n), GetValue(sol, 2 * i + 1 + 2 * n)};
229 
230  FloatType di[2] = {aI[0] * cos(params::omega * t0) + bi[0] * sin(params::omega * t0),
231  aI[1] * cos(params::omega * t0) + bi[1] * sin(params::omega * t0)};
232 
233  FloatType vi[2] = {params::omega * (bi[0] * cos(params::omega * t0)
234  - aI[0] * sin(params::omega * t0)) / params::c,
235  params::omega * (bi[1] * cos(params::omega * t0)
236  - aI[1] * sin(params::omega)) / params::c};
237  FloatType ai[2] = {-pow(params::omega, 2) * di[0], -pow(params::omega, 2) * di[1]};
238 
239  FloatType vsi = vi[0] * s[0] + vi[1] * s[1];
240  FloatType asi = ai[0] * s[0] + ai[1] * s[1];
241 
242  for (int coord = 0; coord < 2; ++coord) {
243  FloatType ttt = ai[coord] * (vsi * sin(theta) - 1) +
244  s[coord] * asi * pow(sin(theta), 2) -
245  vi[coord] * asi * sin(theta);
246  ress[coord] += ttt;
247  }
248 
249  FloatType t3 = asi * sin(theta) * cos(theta);
250  ress[2] += t3;
251  }
252  for (FloatType elem : ress) {
253  res += elem * elem;
254  }
255  return res;

363  d2 = xi[1][i1] - xi[1][i2];
