158 void Dipoles::setMatrixes(const Container &xi) {
159 std::vector<std::pair<int, int>> sectors(4);
160 sectors[0] = {0, 0};
161 sectors[1] = {0, 2 * N_};
162 sectors[2] = {2 * N_, 0};
163 sectors[3] = {2 * N_, 2 * N_};
164 for (int I = 0; I < N_; ++I) {//MAC
165 for (int M = 0; M < N_; ++M) {
166 if (I == M) {
167 Eigen::Matrix<FloatType, 2, 2> id = Eigen::Matrix<FloatType, 2, 2>::Identity() *
168 (params::omega0 * params::omega0 -
169 params::omega * params::omega);
170 M1_.template block<2, 2>(2 * I, 2 * M) = id;
171 } else {
172
173 Eigen::Vector<FloatType, 2> rim = twoDVecDifference(M, I, xi);
174 auto rMode = getDistance(I, M, xi);
175 Eigen::Matrix<FloatType, 2, 2> K1;
176 Eigen::Matrix<FloatType, 2, 2> K2;
177 getMatrices(rim, rMode, K1, K2);
178 auto arg = params::omega * rMode / params::c;
179 Eigen::Matrix<FloatType, 2, 2> tmpmatr = -an * (K1 * cos(arg) - K2 * sin(arg));
180 M1_.template block<2, 2>(2 * I, 2 * M) = tmpmatr;
181 }
182 }
183 }
184
185 for (int I = 0; I < N_; ++I) {//MBC
186 for (int M = 0; M < N_; ++M) {
187 if (I == M) {
188 Eigen::Matrix<FloatType, 2, 2> id =
189 Eigen::Matrix<FloatType, 2, 2>::Identity() * (params::yo * params::omega);
190 M2_.template block<2, 2>(2 * I, 2 * M) = -id;
191 } else {
192
193 Eigen::Vector<FloatType, 2> rim = twoDVecDifference(M, I, xi);
194 auto rMode = getDistance(I, M, xi);
195 Eigen::Matrix<FloatType, 2, 2> K1;
196 Eigen::Matrix<FloatType, 2, 2> K2;

198 auto arg = params::omega * rMode / params::c;

222 void Dipoles::getFullFunction_(const Container &xi, const Container2 &sol) {
223 this->Ifunction_ = [&xi, &sol](FloatType theta, FloatType phi, FloatType t) {
224 int N = xi.size() / 2;
225 FloatType res = 0;
226 FloatType s[2] = {cos(phi), sin(phi)};
227 FloatType ress[3] = {0, 0, 0};
228 for (int i = 0; i < N; ++i) {
229 FloatType ri[2] = {getElement(xi, 0, i, N),
230 getElement(xi, 1, i, N)};
231 FloatType ys = (ri[1] * cos(phi) - ri[0] * sin(phi)) * sin(theta);
232 FloatType t0 = t - ys / params::c;
233 FloatType Ai[2] = {get_value(sol, 2 * i), get_value(sol, 2 * i + 1)};
234 FloatType Bi[2] = {get_value(sol, 2 * i + 2 * N), get_value(sol, 2 * i + 1 + 2 * N)};
235
236
237 FloatType Di[2] = {Ai[0] * cos(params::omega * t0) + Bi[0] * sin(params::omega * t0),
238 Ai[1] * cos(params::omega * t0) + Bi[1] * sin(params::omega * t0)};
239
240 FloatType vi[2] = {params::omega * (Bi[0] * cos(params::omega * t0)
241 - Ai[0] * sin(params::omega * t0)) / params::c,
242 params::omega * (Bi[1] * cos(params::omega * t0)
243 - Ai[1] * sin(params::omega)) / params::c};
244 FloatType ai[2] = {-pow(params::omega, 2) * Di[0], -pow(params::omega, 2) * Di[1]};
245
246 FloatType vsi = vi[0] * s[0] + vi[1] * s[1];
247 FloatType asi = ai[0] * s[0] + ai[1] * s[1];
248
249 for (int coord = 0; coord < 2; ++coord) {
250 FloatType ttt = ai[coord] * (vsi * sin(theta) - 1) +
251 s[coord] * asi * pow(sin(theta), 2) -
252 vi[coord] * asi * sin(theta);
253 ress[coord] += ttt;
254 }
255
256 FloatType t3 = asi * sin(theta) * cos(theta);
257 ress[2] += t3;
258 }
259 for (FloatType elem: ress) {
260 res += elem * elem;
261 }
262 return res;
263 };

355 + 2 * Pcomi.cwiseProduct(Pcomi)) / 2.0;
356 resxy += ri_xy;

359 resz += ri_xz;
360 }
361 resxy = resxy * T0 * o2 * o2;
362 resz = resz * T0 * (o2 * o2 * pow(sin(theta) * cos(theta), 2));
363 res = resxy.sum() + resz;

372 FloatType Dipoles::getDistance(size_t i1, size_t i2, Container &xi) {
373 FloatType d1;
374 FloatType d2;
375 if constexpr (myConcepts::HasBracketsNested<Container>) {
376 d1 = xi[0][i1] - xi[0][i2];
377 d2 = xi[1][i1] - xi[1][i2];
378 } else {
379 d1 = xi[i1] - xi[i2];
380 d2 = xi[i1 + N_] - xi[i2 + N_];
381 }
382 return std::sqrt(d1 * d1 + d2 * d2);
383 }
